1. 快速幂模板

   ```c++
   #include <bits/stdc++.h>
   using namespace std;
   int main(){
       long long a,b;//计算a^b^
       cin >> a >> b;
       long long res = 1;
       while(b){
           if(b & 1) res *= a;
           a *= a;
           b >>= 1;
       }
       cout << res;
   }
   ```

   ```c++
   #include <bits/stdc++.h>
   using namespace std;
   int main(){
       long long a,b,p;//计算a^b^modp
       cin >> a >> b >> p;
       long long res = 1;
       while(b){
           if(b & 1) res = res * a % p;
           a = a * a % p;
           b >>= 1;
       }
       cout << res;
   }
   ```

   八数之和

   计算(1^8^ + 2^8^ + 3^8^ + ···+n^8^) mod 123456789

   ```c++
   #include "bits/stdc++.h"
   using namespace std;
   using ll = long long;
   ll fastPower(ll a,ll b,ll p){
     ll res = 1;
     while(b){
       if(b & 1) res = (res * a) % p;
       a = (a * a) % p;
       b = b >> 1;
     }
     return res;
   }
   int main(){
     int N; cin >> N;
     ll res = 0;
     for(int i = 1 ; i <= N ; i ++){
       res = (res + fastPower(i,8,123456789)) % 123456789;
     }
     cout << res;
     return 0;
   }
   ```

   

2. GCD(a,b) LCM(a,b)

   ```c++
   int GCD(a,b){
       if(a==0) return b;
       return GCD(b%a,a);
   }
   int LCM(a,b){
       return a * b / GCD(a,b);
   }
   ```

3. A C

   A_{N}^{M} = \frac{N!}{(N-M)!}\\
   C_{N}^{M} = \frac{N!}{(N-M)!M!}\\
   C_{N}^{K} = \frac{N-K+1}{K}C_{N}^{K-1}
   $$
