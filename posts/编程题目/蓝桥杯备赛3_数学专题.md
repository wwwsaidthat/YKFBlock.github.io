1. 快速幂模板

   ```c++
   #include <bits/stdc++.h>
   using namespace std;
   int main(){
       long long a,b;//计算a^b^
       cin >> a >> b;
       long long res = 1;
       while(b){
           if(b & 1) res *= a;
           a *= a;
           b >>= 1;
       }
       cout << res;
   }
   ```

   ```c++
   #include <bits/stdc++.h>

   int main(){
       long long a,b,p;//计算a^b^modp
       cin >> a >> b >> p;
       long long res = 1;
       while(b){
           if(b & 1) res = res * a % p;
           a = a * a % p;

       }

   }
   ```

   八数之和

   计算(1^8^ + 2^8^ + 3^8^ + ···+n^8^) mod 123456789

   ```c++
   #include "bits/stdc++.h"
   using namespace std;
   using ll = long long;
   ll fastPower(ll a,ll b,ll p){
     ll res = 1;
     while(b){
       if(b & 1) res = (res * a) % p;
       a = (a * a) % p;
       b = b >> 1;
     }
     return res;
   }
   int main(){
     int N; cin >> N;
     ll res = 0;
     for(int i = 1 ; i <= N ; i ++){
       res = (res + fastPower(i,8,123456789)) % 123456789;
     }
     cout << res;
     return 0;
   }
   ```

欧拉函数代表1-i有多少个与i互质的数
$$
\phi(n) = n \prod_{i=1}^{k} (1-\frac{1}{p_i})
$$
```c++
#include "bits/stdc++.h"
using namespace std;
using ll = long long;
const int N = 1e7;
const int K = 1e5;
bool f[N+5];
int primes[N+5],cnt = 0;
void findPrime(){
	for(int i = 2 ; i <= N ; i++){
		if(!f[i])primes[++cnt] = i;
		for(int j = 1 ; i * primes[j] <= N; j++){
			f[i * primes[j]] = true;
			if(i % primes[j] == 0) break;
		}
	}
}
int F(int i){
	int ans = i;
	if(i == 1) return 1;
	for(int k = 1 ; primes[k] <= i ; k++){
		if(i % primes[k] == 0) {
		 	ans *= (primes[k] - 1);
		 	ans /= primes[k];
		}
			 
	}
	return ans;
}
int main(){
ios::sync_with_stdio(false); cin.tie(nullptr);
	int t; cin>>t;
	findPrime();
//	cout << F(9);
	while(t--){
		int n; cin>>n;
		int sum = 1;
		for(int i = 2 ; i <= n ; i++ ){
			sum += F(i);
		}
		cout << sum << endl;
	}
}
```

组合数的求法 比较简单的写法
$$
\boxed{C_N^M = C_{N-1}^{M} + C_{N-1}^{M-1}}
$$

```c++
#include "bits/stdc++.h"
using namespace std;
using ll = long long;
ll C(int n,int m){
    if(m==0||m==n) return n;
    else return C(n-1,m)+C(n-1,m-1);
}
int main(){
    int m,n;
    cin>>m>>n;
    cout<<C(m,n);
    return 0;
}
```
