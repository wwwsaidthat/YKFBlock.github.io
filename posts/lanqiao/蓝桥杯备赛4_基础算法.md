1. 二分查找

   1. 最简单的寻找一个排序数组是否有binary_search(arr.begin(),arr.end(),target) 这个只能是找target是否有找不到具体位置 直接用lowerbound会更好一些

   2. 寻找一个排好序的数组某个数的最左边界和最右边界

      auto it1 = lower_bound(vector.begin(),vector.end(),target); // >=

      auto it2 = upper_bound(vector.begin(),vector.end(),target);// >

      如果没找到返回vector.end() 

      脚标是it1-vector.begin(); it2-vector.begin() - 1;

      ```c++
      #include <bits/stdc++.h>
      using namespace std;
      
      int main(){
          ios::sync_with_stdio(false);
          cin.tie(nullptr);
      
          int n,q;
          cin >> n >> q;
          vector<int> arr(n,0);//sorted up
          for(int i = 0; i < n ; i ++) cin >> arr[i];
          int query;
          while(q--){
              cin >> query;
              auto it1 = lower_bound(arr.begin(),arr.end(),query);//>=
              auto it2 = upper_bound(arr.begin(),arr.end(),query);//>
              if(it1 == arr.end()||*it1 != query) cout << -1 <<" "<< -1 << endl; 
              else{
                  cout << it1 - arr.begin() <<" "<< it2 - arr.begin() - 1<<endl;
              }
          }
          
          return 0;
      }
      
      ```

   3. 二分查找的模板--关键点在于check的判断，关键在于单调性，在check里面的条件，必须是关于x增长单调变化的一个量才可以用模板，否则不能用模板。

      ```c++
      int l = 0;
      int r = n;
      while(r - l > 1){
          int mid = (l + r)>>1;
          if(check(mid)) l = mid;
          else r = mid;
      }
      if(check(r)) return r;
      else if(check(l)) return l;
      return -1;
      ```

      寻找满足条件的最大值，动左侧，往大了移动

      ```c++
      int l = 0;
      int r = n;

          if(check(mid)) r = mid;
          else l = mid;
      }

      ```

      寻找满足条件的最小值,动右侧，往小了移动

      砍树

      ```c++
      #include <bits/stdc++.h>
      using namespace std;
      using ll = long long;
      ll n,m;
      ll maxn;
      const int N = 1e6 + 6;
      ll h[N];
      int check(int x){
          ll sum = 0;
          for(int i = 0 ; i < n ; i++){
              if(h[i] > x) sum += h[i] - x;
          }
          return sum >= m;
      }
      int main(){
          ios::sync_with_stdio(false);
          cin.tie(nullptr);
          cin >> n >> m;
          for(int i = 0 ; i < n ; i ++){
             cin >> h[i];
             maxn = max(maxn,h[i]);
          }
          int l = 0 ,r = maxn;
          while(r - l > 1){
              int mid = (r + l) >> 1;
              if(check(mid)) l = mid;
              else r = mid;
          }
          if(check(r)) cout << r;
          else if(check(l)) cout <<l;
          return 0;
      }
      
      ```

      

2. 双指针（尺取）

   1. 去重 int fast = 0;int slow = 0;

      ```c++
      for(int fast = 1; fast < arr.size() ;fast++){
          if(arr[fast] != arr[fast + 1]) arr[++slow] = arr[fast];
      }
      ```

   2. 思路就是满足条件的时候一直移动快指针并维护，不满足条件的时候移动慢指针直到满足条件，再接着移动快指针

      ```c++
      for(快指针动){
          记录并维护
      	while(不符合要求){
              移动慢指针直到符合要求    
          }
      }
      ```

      

3. 前缀和和差分

   1. 前缀和可以快速获得一个区间的和 i到j f[j] - f[i - 1]

   2. 前缀和可以配合map使用，用find这个是O(1)的复杂度

   3. 二维前缀和

      ```c++
      s[i][j] = s[i - 1][j - 1] + a[i][j - 1] + a[i - 1][j] ;
      target = s[i][j] + s[i - k][j] + s[i][j - m] - s[i - k][j - m];
      ```

   4. 差分是用来解决对一个区间内每个数都加或者减去某一个数，把o(n)->o(1)

      ```c++
      d[0] = a[0];
      d[1] = a[1] - a[0];
      d[i] = a[i] - a[i - 1];
      ```

   5. 二维差分 

      ```c++
      d[i][j] = a[i][j] - a[i-1][j] -a[i][j - 1] + a[i - 1][j - 1];
      ```

   6. 变化:

      1. 一维差分开始+k结尾的后一个-k
      2. 二维左上+k 右下外+k 左下外右上外 -k

   7. 恢复

      ```c++
      a[i][j] = d[i][j]+a[i-1][j]+a[i][j - 1] - a[i-1][j-1];
      ```
