# 适配器模式



```java
 class typecCharge{
  public void charge(){
    System.out.println("type-c start charge");
  }
}
 class macCharge{
  public void fastCharge (){
    System.out.println("mac start charge");
  }
}

 class ChargeAdapter extends typecCharge{
  private macCharge mcc;
  
  public ChargeAdapter(macCharge sample){
    this.mcc = sample;
  }
  @Override
  public void charge(){
    mcc.fastCharge();
  }
  
}

public class Main{
  public static void main(String[] args){
    macCharge mcc = new macCharge();
    typecCharge tyc = new ChargeAdapter(mcc);
    tyc.charge();
    
  }
}
```
