# 模版方法模式

```java
public abstract class DbOperator {
    public final void process() {
        openDb();
        connectDb();
        useDb();
        closeDb();
    }

    protected void openDb() {
        System.out.println("打开数据库资源");
    }

    protected abstract void connectDb();

    protected void useDb() {
        System.out.println("使用数据库");
    }

    protected void closeDb() {
        System.out.println("关闭数据库资源");
    }
}

public class SqlServerDbOperator extends DbOperator {
    @Override
    protected void connectDb() {
        System.out.println("连接 SQL Server");
    }
}

public class OracleDbOperator extends DbOperator {
    @Override
    protected void connectDb() {
        System.out.println("连接 Oracle");
    }
}
```
