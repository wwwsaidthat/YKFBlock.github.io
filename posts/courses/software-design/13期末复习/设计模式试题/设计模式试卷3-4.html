C卷 参考答案

一、分析题

1. 违背：里氏替换原则
原因：正方形重写了长方形的宽高设置方法，破坏了父类的行为逻辑，子类不能完全替换父类。
改进：抽取抽象图形类，长方形和正方形分别独立继承，取消继承关系。

2. 适用模式：外观模式
思路：创建外观类封装所有子系统操作，调用方只依赖外观类，降低耦合，统一调用入口。

3. 适配器模式类图

// 目标接口：SD卡读卡器
class SDCard {
public:
    virtual void readSD() = 0;
    virtual ~SDCard() = default;
};
// 适配者：TF卡
class TFCard {
public:
    void readTF() { cout << "读取TF卡数据" << endl; }
};

class Adapter : public SDCard, private TFCard {
public:
    void readSD() override { readTF(); }
};

2. 迭代器模式

// 抽象迭代器
class Iterator {
public:
    virtual int next() = 0;
    virtual bool hasNext() = 0;
    virtual ~Iterator() = default;
};
// 具体集合
class Array {
private:
    int arr[5] = {1,2,3,4,5};
    int index = 0;
public:
    Iterator* createIterator();
    int get(int i) { return arr[i]; }
    int size() { return 5; }
};
// 具体迭代器
class ArrayIter : public Iterator {
private:
    Array* arr;
    int idx = 0;
public:
    ArrayIter(Array* a) : arr(a) {}
    int next() override { return arr->get(idx++); }
    bool hasNext() override { return idx < arr->size(); }
};
Iterator* Array::createIterator() { return new ArrayIter(this); }

#include

class Button { public: virtual void show() = 0; };

}
