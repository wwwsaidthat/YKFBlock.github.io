#include 
using namespace std;
// 抽象策略
class CashStrategy {
public:
    virtual double acceptCash(double money) = 0;
    virtual ~CashStrategy() = default;
};
// 正常收费
class NormalCash : public CashStrategy {
public:
    double acceptCash(double money) override {
        return money;
    }
};
// 打折
class RebateCash : public CashStrategy {
private:
    double rebate;
public:
    RebateCash(double r) : rebate(r) {}
    double acceptCash(double money) override {
        return money * rebate;
    }
};
// 上下文
class CashContext {
private:
    CashStrategy* strategy;
public:
    CashContext(CashStrategy* s) : strategy(s) {}
    double getResult(double money) {
        return strategy->acceptCash(money);
    }
    ~CashContext() { delete strategy; }
};

#include

#include 
using namespace std;
class MenuComponent {
protected:
    string name;
public:
    MenuComponent(string n) : name(n) {}
    virtual void add(MenuComponent*) {}
    virtual void display(int depth) = 0;
    virtual ~MenuComponent() = default;
};
// 菜单项（叶子）
class MenuItem : public MenuComponent {
public:
    MenuItem(string n) : MenuComponent(n) {}
    void display(int depth) override {
        cout << string(depth, '-') << name << endl;
    }
};
// 菜单（容器）
class Menu : public MenuComponent {
private:
    vector vec;
public:
    Menu(string n) : MenuComponent(n) {}
    void add(MenuComponent* m) override {
        vec.push_back(m);
    }
    void display(int depth) override {
        cout << string(depth, '-') << name << endl;
        for(auto c : vec)
            c->display(depth + 2);
    }
    ~Menu() {
        for(auto c : vec) delete c;
    }
};
// 客户端
int main() {
    Menu* root = new Menu("主菜单");
    Menu* file = new Menu("文件");
    file->add(new MenuItem("新建"));
    file->add(new MenuItem("打开"));
    Menu* edit = new Menu("编辑");
    edit->add(new MenuItem("复制"));
    edit->add(new MenuItem("粘贴"));
    root->add(file);
    root->add(edit);
    root->display(0);
    delete root;
    return 0;
}

2. 建造者+代理（电脑+权限）

#include 
#include 
using namespace std;
class Computer {
public:
    string cpu, gpu, mem;
    void show() {
        cout << cpu << "|" << gpu << "|" << mem << endl;
    }
    void powerOn() { cout << "电脑开机" << endl; }
};
class Builder {
protected:
    Computer* c;
public:
    Builder() { c = new Computer; }
    virtual void buildCPU() = 0;
    virtual void buildGPU() = 0;
    virtual void buildMem() = 0;
    Computer* get() { return c; }
    virtual ~Builder() { delete c; }
};
class HighBuilder : public Builder {
public:
    void buildCPU() override { c->cpu = "i9"; }
    void buildGPU() override { c->gpu = "4090"; }
    void buildMem() override { c->mem = "64G"; }
};
// 代理权限
class PowerProxy {
private:
    Computer* comp;
    bool auth;
public:
    PowerProxy(Computer* c, bool a) : comp(c), auth(a) {}
    void powerOn() {
        if(auth) {
            comp->powerOn();
        } else {
            cout << "无权限" << endl;
        }
    }
};
// 客户端
int main() {
    HighBuilder b;
    b.buildCPU(); b.buildGPU(); b.buildMem();
    Computer* pc = b.get();
    pc->show();
    PowerProxy p1(pc, true);
    p1.powerOn();
    PowerProxy p2(pc, false);
    p2.powerOn();
    return 0;
}

class Logger {

    virtual void log(string s) = 0;
    virtual ~Logger() = default;

        log->log("加密:" + s);
    }
};

public:

2. 状态模式+单例（任务调度）

#include 
#include 
using namespace std;
class Task;
class State {
public:
    virtual void handle(Task* t) = 0;
    virtual ~State() = default;
};
class Task {
private:
    State* state;
    static Task* ins;
    Task() : state(nullptr) {}
public:
    static Task* get() {
        if(!ins) ins = new Task;
        return ins;
    }
    void setState(State* s) { state = s; }
    void doJob() { state->handle(this); }
};
Task* Task::ins = nullptr;
// 待执行
class WaitState : public State {
public:
    void handle(Task* t) override {
        cout << "任务待执行" << endl;
    }
};
// 运行中
class RunState : public State {
public:
    void handle(Task* t) override {
        cout << "任务运行中" << endl;
    }
};
// 客户端
int main() {
    Task* t = Task::get();
    t->setState(new WaitState);
    t->doJob();
    t->setState(new RunState);
    t->doJob();
    return 0;
}
